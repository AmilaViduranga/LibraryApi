pipeline {
	agent any
	stages {
		stage('library-app-compile-stage'){
			steps {
				withMaven(maven: 'maven_3_5_4') {
					sh 'mvn clean compile'
				}
			}
		}
		stage('library-app-testing-stage'){
			steps {
				withMaven(maven: 'maven_3_5_4') {
					sh 'mvn test'
				}
			}
		}
		stage('library-app-deployment-stage'){
			steps {
				withMaven(maven: 'maven_3_5_4') {
					sh 'mvn package'
				}
			}
		}
	}
}
